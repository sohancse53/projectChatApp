CREATE DATABASE phpchatapp_db;
CREATE TABLE users(
user_id int(11) PRIMARY KEY AUTO_INCREMENT ,
unique_id VARCHAR(255) DEFAULT NULL,
fname VARCHAR(255) DEFAULT NULL,
lname VARCHAR(255) DEFAULT NULL,
email  VARCHAR(255) DEFAULT NULL,
password VARCHAR(255) DEFAULT NULL,
status VARCHAR(255) 
);
CREATE TABLE messages(
msg_id int(11) PRIMARY KEY AUTO_INCREMENT,
incoming_msg_id VARCHAR(255) DEFAULT NULL,
outgoing_msg_id VARCHAR(255) DEFAULT NULL,
msg VARCHAR(255) DEFAULT NULL);
ALTER TABLE users ADD img VARCHAR(255) DEFAULT NULL;
ALTER TABLE users DROP status;
ALTER TABLE users ADD status VARCHAR(255);
ALTER TABLE `users` CHANGE `status` `status` VARCHAR(255) CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci NULL DEFAULT NULL COMMENT 'Online now , Offline';
